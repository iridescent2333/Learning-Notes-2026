### 存储结构
-     /**
      * Definition for singly-linked list.
      * struct ListNode {
      *     int val;
      *     struct ListNode *next;
      * };
      */
      struct ListNode* mergeTwoLists(struct ListNode* list1, struct ListNode* list2) {
      // 虚拟头结点
      struct ListNode dummy;
      dummy.val = 0;
      dummy.next = NULL;

      struct ListNode* tail = &dummy;

      // 合并过程
      while (list1 != NULL && list2 != NULL) {
          if (list1->val <= list2->val) {
              tail->next = list1;
              list1 = list1->next;
          } else {
              tail->next = list2;
              list2 = list2->next;
          }
          tail = tail->next;
      }

      // 接上剩余部分
      if (list1 != NULL) {
          tail->next = list1;
      } else {
          tail->next = list2;
      }

      return dummy.next;
      }
### 逻辑结构
- 链表问题用哑结点(dummy node)解决，先建立并初始化一个虚拟头结点，再用一个中间结点指向哑结点的地址，并移动中间结点构造出目标链表，合并过程就是while循环遍历谁小接谁，再把剩余部分接上，return哑结点的指针
