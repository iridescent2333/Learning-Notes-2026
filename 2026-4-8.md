#### 存储结构(c语言)
-     //算法实现
      #define MAX 100
      #define max(a,b) a>b?a:b
      int D[MAX][MAX];
      int Filld(int A[],int m,int B[],int n){
        int i,j;
        for(i=1;i<=m;i++)
          for(j=1;j<=n;j++)
            if(A[i]==B[j])
              D[i][j]=D[i-1][j-1]+1;
            else
              D[i][j]=max(D[i][j-1],D[i-1][j]);
        return 1;
      }
-     //算法回溯
      int Seq(int A[],int m,int B[],int n,int C[]){
      int i,j,k;
      i=m;j=n;
      k=D[m][n];
      while(i!=0 && j!=0){
        if(D[i][i]==D[i-1][j]) i--;
         else if(D[i][j] == D[i][j-1]) j--;
         else{
                  C[k--]=A[i];
                  i--;j--;
         }
      }
       return D[m][n];
      }

-     int longestCommonSubsequence(char* text1, char* text2) {
      #define MAX 100
        int S[MAX][MAX];
        int i, j;

        int len1 = 0, len2 = 0;
        while (text1[len1] != '\0') len1++;
        while (text2[len2] != '\0') len2++;

        // 初始化边界
        for (i = 0; i <= len1; i++) S[i][0] = 0;
        for (j = 0; j <= len2; j++) S[0][j] = 0;

        // DP 填表
        for (i = 1; i <= len1; i++) {
              for (j = 1; j <= len2; j++) {
                  if (text1[i-1] == text2[j-1]) {
                      S[i][j] = S[i-1][j-1] + 1;
                  } else {
                      S[i][j] = (S[i-1][j] > S[i][j-1] ? S[i-1][j] : S[i][j-1]);
                  }
              }
        }

          return S[len1][len2];
      }
#### 逻辑结构
- 简单定义max函数，再依次循环遍历两个数组的所有元素，相同则对结果数组进行对角项前一个元素加1操作赋值，不同则取上方或者左方前一个元素的较大值的操作赋值
- 回溯则时从结果数组出发，依次遍历寻找左方和上方相同元素，并将二维数组index减1，输出值为长度
