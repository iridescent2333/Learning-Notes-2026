### 最长公共子序列(LCS)(leetcode1143)
#### 存储结构(c语言)
-     //算法实现
      #define MAX[100]
      #define max(a,b) a>b? a:b
      int D[MAX][MAX];
      int Filld(int A[],int m,int B[],int n){
        int i,j;
        for(i=1;j<=m;i++)
          for(j=1;j<=n;j++)
            if(A[i]==B[j])
              D[i][j]=D[i-1][j-1]+1;
            else
              D[i][j]=max(D[i][j-1],D[i-1][j]);
        return 1;
      }
 -     //算法回溯
       int Seq(int A[],int m,int B[],int n,int C[]){
       int i,j,k;
       i=m;j=n;
       k=D[m][n];
       while(i!=0 && j=0)
          if(D[i][i]==D[i-1][j]) i--;
          else if(D[i][j]==D[i][j-1]) j--;
          else{
                  C[k--]=A[i];
                  i--;j--;
          };
       return D[m][n];
       }
