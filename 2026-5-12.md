## 洛谷刷题(p1440，)(类似力扣刷题239，滑动窗口最大值)
### 求m区间内的最小值
- 单调队列，滑动窗口极值问题，
-     #include <stdio.h>
      #define MAXN 2000010
      // 单调队列：存储数组下标，保证对应的值单调递增
      int q[MAXN];
      int a[MAXN];
      int head, tail;

      int main() {
          int n, m;
          // 加速输入输出（大数据必备）
          scanf("%d%d", &n, &m);
          for (int i = 1; i <= n; i++) {
              scanf("%d", &a[i]);
          }

          head = 1;
          tail = 0;

          for (int i = 1; i <= n; i++) {
              // 第一步：输出答案
              if (i == 1) {
                  printf("0\n");
              } else {
                  printf("%d\n", a[q[head]]);
              }

              // 第二步：维护队列单调性：比当前数大的都弹出
              while (head <= tail && a[i] <= a[q[tail]]) {
                  tail--;
              }
              tail++;
              q[tail] = i;

              // 第三步：超出窗口范围的弹出
              while (q[head] <= i - m) {
                  head++;
              }
          }
          return 0;
      }
