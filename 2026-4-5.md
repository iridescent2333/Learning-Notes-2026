## MySQL数据库通用语法
### DDL
#### 查询所有数据库
-     show databases;
#### 查询当前数据库
-     select database();
#### 创建数据库
-     create database if not exists A default charset utf8 collate True;
-     --eg:创建名为A的数据库，指定字符集为utf8
      create database A charset = utf8;
#### 删除数据库
-     drop database if exists A;
#### 使用数据库
-     use A;
#### 查询当前数据库所有表
-     show tables;
#### 查询表结构
-     desc a;
#### 查询指定表的建表语句
-     show create table a;
#### 表操作-创建
-     create table a(
          字段1 字段1类型 comment 字段1注释,
          字段2 字段2类型 comment 字段2注释,
          ......
          字段n 字段n类型 comment 字段n注释
      )comment 表注释;
#### 表操作-修改
##### 添加字段
-     alter table a add 字段名 类型 comment 注释 约束;
##### 修改数据类型
-     alter table a modify 字段名 新数据类型;
##### 修改字段名和数据类型
-     alter table a change 旧字段名 新字段名 类型 comment 注释 约束;
##### 删除字段
-     alter table a drop 字段名;
##### 修改表名
-     alter table a rename to b;
#### 表操作-删除
##### 删除表
-     drop table if exists a;
##### 删除指定表，并重新创建该表
-       truncate table a;
### DML
#### 添加数据
##### 给指定字段添加数据
-       insert into a(字段名1,字段名2,...) values (值1，值2,...);
##### 给全部字段添加数据
-       insert into a values (值1，值2,...);
##### 批量添加数据
-       insert into a(字段名1,字段名2,...) values (值1，值2,...),(值1，值2,...),(值1，值2,...);
-       insert into a values (值1，值2,...),(值1，值2,...),(值1，值2,...);
#### 修改数据
-       update a set 字段名1=值1,字段名2=值2,... where 条件;
#### 删除数据
-       delete from a where 条件;
### DQL
#### 基本查询
##### 查询多个字段
-       select 字段1,字段2,... from a;
-       select * from a;
##### 设置别名
-       select 字段1 as 别名1,字段2 as 别名2,... from 表名;
##### 去除重复记录
-       select distinct 字段列表 from a;
#### 条件查询
-       select 字段列表 from a where 条件列表;
-       --between ... and ...
-       --in(...)
-       --like 占位符   模糊匹配(_匹配单个字符,%匹配任意个字符)
-       --is null
