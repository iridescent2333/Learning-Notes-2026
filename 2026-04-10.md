        int i,p=0;
        for(i=1;i<n;i++)
          if(A[i]>A[p]) p=i;
      return p;
      }

      int i;
      *end=Maxp(A,n);

-     //力扣习题53
      int maxSubArray(int* nums, int numsSize) {
      if (numsSize == 0) return 0;

      int S[numsSize];  // C99 变长数组
      int maxSum = nums[0];

      S[0] = nums[0];

      for (int i = 1; i < numsSize; i++) {
          S[i] = (S[i - 1] > 0) ? S[i - 1] + nums[i] : nums[i];
          if (S[i] > maxSum) {
              maxSum = S[i];
          }
      }

      return maxSum;
      }
### 逻辑结构
- 定义一个结果数组，循环遍历原数组，以结果数组的前一项的值是否大于零为判断标准，给结果数组赋值为结果数组前一项加原数组此项的和或从原数组此项重新开始计算，最终遍历结果数组，最大值即为最大子段和
